	PersonligVirksomhedOplysning deles i to:
		(
			*PersonligVirksomhedOplysning*
			[
				(SelvstændigVirksomhedOverskud)
				(SelvstændigVirksomhedUnderskud)
				(VirksomhedTypeLigningArtKode)
				(VirksomhedTypeFremtidigLigningArtKode)
				(VirksomhedOplysningSamledeHævningerBeløb)
			]
		)
		*RegnskabOplysningListe*
		0{
			*RegnskabOplysning*
			[
				(
					*RegnskabsoplysningValg*
					[
						VirksomhedCVRNummer
						|
						PersonCPRNummer
					]
				)
			(RegnskabOplysningNettoomsætning)
			(RegnskabOplysningOrdinærtResultatBeløb)
			]
		}
